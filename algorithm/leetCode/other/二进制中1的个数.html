      # 二进制中1的个数

      链接open in new window

      n & (n - 1)可以消除n最右边的一个 1（二进制表示）

      如：6 & (6−1)=4, 6=(110)2, 4=(100)2, 6&(6-1) = 4, 运算结果 4 即为把 6 的二进制位中的最低位的 1 变为 0之后的结果。

      var hammingWeight = function(n) {
    let count = 0;
   while(n) {
    
       n = n & (n-1);
       count ++;
   }
   return count;
};

      var hammingWeight = function(n) {
    let count = 0;
    // 转换为二进制字符串数组
    n = n.toString(2).split('');
   for(let i of n) {
       if(i == 1) {
           count ++;
       }
   }
    return count;
};
