                              # 链式调用

                              class Man {
  constructor() {
    this.queue = [];
    this.index = 0;
    this.firstSleepWatch = false;
    this.init();
  }

  init() {
    setTimeout(() => {
      this.run();
    })
  }

  run() {
    const fn = this.queue[this.index];
    this.index++;
    fn && fn();
  }

  firstSleep(delay) {
    if (this.firstSleepWatch) {
     throw new Error('Already declared firstSleep');
    }
    this.queue.unshift(() => {
      setTimeout(() => {
        console.log(`${delay} firstSleep`);
        this.run();
      }, delay)
    })
    return this;
  } 

  sleep(delay) {
    this.queue.push(() => {
      setTimeouut(() => {
        console.log(`${delay} sleep end`);
        this.run()      
      },delay)
    })
    return this;
  }

  work() {
    this.queue.push(() => {
      console.log('work running');
      this.run();
    })
    return this;
  }
}
